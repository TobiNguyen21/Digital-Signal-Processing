import matplotlib.pyplot as plt 
from scipy import signal
import numpy as np 
from scipy import fftpack as fft

Ts = 1/140;
t = np.arange(0,5,Ts); f1 =2; f2 = 6; f3 = 10; f4 = 14;
s = 2*np.sin(2*np.pi*f1*t) + 1*np.sin(2*np.pi*f2*t) + 1*np.sin(2*np.pi*f3*t) + 2*np.sin(2*np.pi*f4*t);

sL = 2*np.sin(2*np.pi*f1*t);
sH = 2*np.sin(2*np.pi*f4*t);
sP = 1*np.sin(2*np.pi*f2*t) + 1*np.sin(2*np.pi*f3*t);
sS = 2*np.sin(2*np.pi*f1*t) + 2*np.sin(2*np.pi*f4*t);
fc1 = 4;  
fc2 = 12;
#========================Dap ung DAI QUA ly tuong===================
N = 81; k = (N-1)/2;
OmegaC1 = 2*np.pi*fc1*Ts;
OmegaC2 = 2*np.pi*fc2*Ts;
hnk = np.zeros(N,dtype=complex);
for i in range (0,N,1):
	if i == k:
		hnk[i] =  1-(OmegaC2)/np.pi+(OmegaC1)/np.pi;
	else: 
		hnk[i] = (OmegaC1/np.pi) * np.sin(OmegaC1*(i-k))/(OmegaC1*(i-k))-(OmegaC2/np.pi) * np.sin(OmegaC2*(i-k))/(OmegaC2*(i-k));
#-----------------------------------------------------------
#============================Cua So ========================
#======================== Cua so hinh chu nhat =============
wR = np.concatenate([np.ones(N)]);


hn = wR* hnk;
#======================== Kiem tra =========================
a = [1];
w, H = signal.freqz(hn, worN = 512, whole = False);

f = w/(2*np.pi*Ts);
Habs = abs(H);
Hpha = np.angle(H);

y = signal.convolve(s,hn);
y = np.real(y);



#==================
plt.subplot(4,1,1);
plt.plot(s);
plt.xlabel('t');
plt.ylabel('s');

plt.subplot(4,1,2);
plt.plot(y);
plt.xlabel('t');
plt.ylabel('y_loc');

plt.subplot(4,1,3);
plt.plot(sS);
plt.xlabel('t');
plt.ylabel('y_LyTuong');

plt.subplot(4,1,4);
plt.plot(f,Habs);
plt.xlabel('f');
plt.ylabel('Habs');


plt.show()
