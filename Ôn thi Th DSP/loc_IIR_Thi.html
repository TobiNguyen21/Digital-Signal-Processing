import matplotlib.pyplot as plt 
from scipy import signal
import numpy as np 
from scipy import fftpack as fft


import math
import sympy as sym

 
#======================= Gia thiet ========================
f1 = 2;  f2 = 4;  f3 = 7; f4 = 9; f5 = 12;
fs = 120; Ts = 1/120;
t = np.arange(0,5,Ts);
st = np.sin(2*np.pi*f1*t) + 2*np.sin(2*np.pi*f2*t) + np.sin(2*np.pi*f3*t) + 2*np.sin(2*np.pi*f4*t) + np.sin(2*np.pi*f5*t);

#----------------------- Tin hieu ly tuong ----------------
sLP = np.sin(2*np.pi*f1*t);
sHP = np.sin(2*np.pi*f5*t);
sBP = 2*np.sin(2*np.pi*f2*t) + np.sin(2*np.pi*f3*t);
sSP = np.sin(2*np.pi*f1*t) + np.sin(2*np.pi*f5*t);

 

#================= Butterworth Thap Qua ===================
SigmaP = 0.1;              SigmaS = 0.1;
fP =2.5;                  fS = 3.5; 
#đối với thấp qua thì fp=2.5    fs=3.5
#đối với cao qua thì  fp=2.5  fs=3;
#đối với thông qua thì fp=1      fs=2
#đối với chặn qua thì fp=2      fs=3

OmegaP = 2*np.pi*fP;       OmegaS = 2*np.pi*fS;
 
 

#----------------------- Bac loc N ------------------------
temp1 = (1-(1-SigmaP)**2) * SigmaS**2;
temp2 = (1-SigmaP)**2 * (1-SigmaS**2);
N = 0.5* math.log10(temp1/temp2) / math.log10(OmegaP/OmegaS);
N = int(N)+1;
 

#----------------------- OmegaC----------------------------
temp1 = (1-(1-SigmaP)**2) / ((1-SigmaP)**2);
OmegaCL = OmegaP/(temp1**(1/(2*N)));
fCL = OmegaCL/(2*np.pi);


#------------------------ pa ------------------------------
pa = np.zeros(N,dtype=complex);
for i in range(0,N,1):
    pa[i] = OmegaCL * np.exp(1j*(2*(i+1)+N-1)*np.pi/(2*N));
 

#--------------- Hs - Thap qua ----------------------------
s = sym.Symbol('s');
w = sym.Symbol('w');
z = sym.Symbol('z');
temp1 = 1;
for i in range(0,N,1):
    temp1 = temp1*(s-pa[i]);

 

HasL = (OmegaCL**N)/temp1;
HasL = sym.lambdify(s,HasL,'numpy');
HawL = HasL(1j*w);
HawL = sym.lambdify(w,HawL,'numpy');


#----------------- Do thi Loc - Dap ung bien do -------------
f = np.arange(0.001,20,0.2);
w_val = 2*np.pi*f;
HawL_val = HawL(w_val);
HawL_abs = abs(HawL_val);

 

s_temp = 2*(1-z)/(Ts*(1+z));
Haz = HasL(s_temp);             #the s vao ha(s) chuyen thanh H(z)
Haz = sym.simplify(Haz);        
Tuso, Mauso = sym.fraction(Haz);
Tuso = sym.expand(Tuso);
Mauso = sym.expand(Mauso);
    


Mauso = sym.poly(Mauso,z);
HesoA = Mauso.coeffs();
HesoA = np.flip(HesoA);
a = HesoA/HesoA[0];
for i in range(0,len(a),1):
    a[i] = np.complex128(a[i]);
    a[i] = np.real(a[i]);


Tuso = sym.poly(Tuso,z);
HesoB = Tuso.coeffs();
HesoB = np.flip(HesoB);
b = HesoB/HesoA[0];
for i in range(0,len(b),1):
    b[i] = np.complex128(b[i]);
    b[i] = np.real(b[i]);


y = signal.lfilter(b,a,st);


plt.subplot(4,1,1)
plt.plot(st);
plt.xlabel('t'); 
plt.ylabel('st');


plt.subplot(4,1,2)
plt.plot(sLP);
plt.xlabel('t'); 
plt.ylabel('s_Lytuong');

 
plt.subplot(4,1,3)
plt.plot(y);
plt.xlabel('t'); 
plt.ylabel('s_Loc');


plt.subplot(4,1,4)
plt.plot(f,HawL_abs);
plt.xlabel('f'); 
plt.ylabel('|H|');

plt.show()
